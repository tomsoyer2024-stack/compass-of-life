const CACHE_NAME = 'compass-v1';
const urlsToCache = [
    '/',
    '/index.html',
    '/manifest.json',
    '/favicon.ico'
];

self.addEventListener('install', event => {
    event.waitUntil(
        caches.open(CACHE_NAME)
            .then(cache => cache.addAll(urlsToCache))
    );
});

self.addEventListener('fetch', event => {
    event.respondWith(
        caches.match(event.request)
            .then(response => response || fetch(event.request))
    );
});

import React, { useState, useEffect, useRef } from 'react';
import { useTranslation } from 'react-i18next';
import { theme } from '../theme';

export default function AIChat({ context, onClose, onApplySteps, character = 'energetic' }) {
    const { t } = useTranslation();

    const getInitialMessage = () => {
        return t(`ai.greeting_${character}`, { context });
    };

    const [messages, setMessages] = useState([
        { role: 'ai', text: getInitialMessage() }
    ]);
    const [input, setInput] = useState('');
    const [isTyping, setIsTyping] = useState(false);
    const scrollRef = useRef(null);

    useEffect(() => {
        if (scrollRef.current) {
            scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
        }
    }, [messages]);

    const handleSend = () => {
        if (!input.trim()) return;
        const userMsg = { role: 'user', text: input };
        setMessages(prev => [...prev, userMsg]);
        setInput('');
        setIsTyping(true);

        // Simulate professional AI response logic with localization
        setTimeout(() => {
            let aiText = t('ai.default_reply');
            let suggestedSteps = null;

            const lowInput = input.toLowerCase();
            const keywords = {
                plan: ['разбей', 'план', 'как', 'plan', 'how', 'breakdown'],
                tired: ['лень', 'устал', 'lazy', 'tired', 'fatigue']
            };

            if (keywords.plan.some(k => lowInput.includes(k))) {
                aiText = t('ai.plan_intro');
                suggestedSteps = [
                    t('common.step_draft_info') || "Collect information (30 min)",
                    t('common.step_prep_space') || "Prepare workspace",
                    t('common.step_first_draft') || "Make first draft",
                    t('common.step_review_next') || "Review results and plan next"
                ];
            } else if (keywords.tired.some(k => lowInput.includes(k))) {
                aiText = t('ai.tired_reply');
            }

            setMessages(prev => [...prev, { role: 'ai', text: aiText, steps: suggestedSteps }]);
            setIsTyping(false);
        }, 1500);
    };

    return (
        <div style={{ display: 'flex', flexDirection: 'column', height: '100%', padding: '10px' }}>
            <div
                style={{
                    flex: 1, overflowY: 'auto', padding: '16px', background: '#f5f7f9',
                    borderRadius: '24px', marginBottom: '16px', border: '1px solid #edf2f7'
                }}
                ref={scrollRef}
            >
                {messages.map((m, i) => (
                    <div key={i} style={{
                        display: 'flex',
                        flexDirection: 'column',
                        alignItems: m.role === 'user' ? 'flex-end' : 'flex-start',
                        marginBottom: '16px'
                    }}>
                        <div style={{
                            maxWidth: '85%',
                            padding: '14px 18px',
                            borderRadius: '20px',
                            background: m.role === 'user' ? theme.colors.ui.primary : 'white',
                            color: m.role === 'user' ? 'white' : theme.colors.text.primary,
                            boxShadow: m.role === 'ai' ? '0 4px 12px rgba(0,0,0,0.03)' : 'none',
                            fontSize: '15px',
                            lineHeight: '1.5',
                            borderBottomRightRadius: m.role === 'user' ? '4px' : '20px',
                            borderBottomLeftRadius: m.role === 'ai' ? '4px' : '20px',
                        }}>
                            {m.text}

                            {m.steps && (
                                <div style={{ marginTop: '12px', borderTop: '1px solid #eee', paddingTop: '10px' }}>
                                    {m.steps.map((step, idx) => (
                                        <div key={idx} style={{ marginBottom: '6px', fontSize: '13px', display: 'flex', alignItems: 'center', gap: '8px' }}>
                                            <div style={{ width: '4px', height: '4px', borderRadius: '50%', background: theme.colors.ui.primary }}></div>
                                            {step}
                                        </div>
                                    ))}
                                    <button
                                        onClick={() => onApplySteps && onApplySteps(m.steps)}
                                        style={{
                                            marginTop: '10px', width: '100%', padding: '8px',
                                            background: theme.colors.ui.primary, color: 'white',
                                            border: 'none', borderRadius: '10px', fontSize: '12px',
                                            fontWeight: 600, cursor: 'pointer'
                                        }}
                                    >
                                        {t('ai.apply_steps')}
                                    </button>
                                </div>
                            )}
                        </div>
                    </div>
                ))}
                {isTyping && <div style={{ color: '#aaa', fontSize: '12px', marginLeft: '12px', fontStyle: 'italic' }}>{t('ai.typing')}</div>}
            </div>

            <div style={{ display: 'flex', gap: '10px', alignItems: 'center' }}>
                <input
                    value={input}
                    onChange={e => setInput(e.target.value)}
                    onKeyDown={e => e.key === 'Enter' && handleSend()}
                    placeholder={t('ai.input_placeholder')}
                    style={{
                        flex: 1, padding: '14px 20px', borderRadius: '30px',
                        border: '1px solid #e2e8f0', outline: 'none', fontSize: '15px',
                        background: 'white'
                    }}
                />
                <button
                    onClick={handleSend}
                    style={{
                        width: '46px', height: '46px', borderRadius: '50%',
                        background: theme.colors.ui.primary, color: 'white',
                        border: 'none', cursor: 'pointer', display: 'flex',
                        alignItems: 'center', justifyContent: 'center',
                        boxShadow: '0 4px 10px rgba(0,0,0,0.1)'
                    }}
                >
                    <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5" strokeLinecap="round" strokeLinejoin="round"><line x1="22" y1="2" x2="11" y2="13"></line><polygon points="22 2 15 22 11 13 2 9 22 2"></polygon></svg>
                </button>
            </div>
        </div>
    );
}

import React, { useState, useEffect, useRef } from 'react';
import { useTranslation } from 'react-i18next';
import { motion, AnimatePresence } from 'framer-motion';
import { theme } from '../theme';

const ELEMENT_TYPES = {
    IMAGE: 'image',
    NOTE: 'note',
    SHAPE: 'shape'
};

const BoardElement = ({ element, onUpdate, onDelete, isEditMode }) => {
    const { t } = useTranslation();
    const [isResizing, setIsResizing] = useState(false);

    const handleDragEnd = (event, info) => {
        onUpdate(element.id, { x: element.x + info.offset.x, y: element.y + info.offset.y });
    };

    return (
        <motion.div
            drag={isEditMode}
            dragMomentum={false}
            onDragEnd={handleDragEnd}
            initial={{ scale: 0.8, opacity: 0 }}
            animate={{ scale: 1, opacity: 1, x: element.x, y: element.y }}
            style={{
                position: 'absolute',
                width: element.width || 120,
                height: element.height || 120,
                cursor: isEditMode ? 'move' : 'default',
                zIndex: isEditMode ? 10 : 1,
            }}
        >
            <div style={{
                width: '100%',
                height: '100%',
                borderRadius: element.type === ELEMENT_TYPES.IMAGE ? '16px' : '8px',
                overflow: 'hidden',
                boxShadow: theme.shadows.card,
                background: element.type === ELEMENT_TYPES.NOTE ? '#FFF176' : '#fff',
                position: 'relative',
                border: isEditMode ? `2px solid ${theme.colors.ui.primary}44` : 'none',
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center',
                padding: element.type === ELEMENT_TYPES.NOTE ? '12px' : '0'
            }}>
                {element.type === ELEMENT_TYPES.IMAGE ? (
                    <img src={element.content} alt="Dream" style={{ width: '100%', height: '100%', objectFit: 'cover' }} />
                ) : (
                    <div style={{ fontSize: '14px', fontWeight: 500, color: '#333', textAlign: 'center', overflow: 'hidden' }}>
                        {element.content}
                    </div>
                )}

                {isEditMode && (
                    <button
                        onClick={() => onDelete(element.id)}
                        style={{ position: 'absolute', top: 5, right: 5, background: '#ff4d4d', color: '#fff', border: 'none', borderRadius: '50%', width: 24, height: 24, cursor: 'pointer', fontSize: 12, fontWeight: 800, display: 'flex', alignItems: 'center', justifyContent: 'center' }}
                    >
                        ✕
                    </button>
                )}
            </div>
        </motion.div>
    );
};

export default function Board() {
    const { t } = useTranslation();
    const [elements, setElements] = useState(() => {
        try {
            const saved = localStorage.getItem('dream_board_elements');
            return saved ? JSON.parse(saved) : [];
        } catch { return []; }
    });
    const [mode, setMode] = useState('vision'); // vision, work
    const [isEditMode, setIsEditMode] = useState(true);
    const boardRef = useRef(null);

    useEffect(() => {
        localStorage.setItem('dream_board_elements', JSON.stringify(elements));
    }, [elements]);

    const addElement = (type, content = '') => {
        const newEl = {
            id: Date.now().toString(),
            type,
            content: content || (type === ELEMENT_TYPES.NOTE ? 'New Note' : ''),
            x: 50 + Math.random() * 100,
            y: 50 + Math.random() * 100,
            width: 150,
            height: 150,
            rotate: 0
        };
        setElements([...elements, newEl]);
    };

    const updateElement = (id, updates) => {
        setElements(elements.map(el => el.id === id ? { ...el, ...updates } : el));
    };

    const deleteElement = (id) => {
        setElements(elements.filter(el => el.id !== id));
    };

    const handleImageUpload = (e) => {
        const file = e.target.files[0];
        if (file && file.size <= 2 * 1024 * 1024) {
            const reader = new FileReader();
            reader.onloadend = () => addElement(ELEMENT_TYPES.IMAGE, reader.result);
            reader.readAsDataURL(file);
        } else if (file) {
            alert(t('board.upload_help'));
        }
    };

    return (
        <div style={{ height: 'calc(100vh - 160px)', display: 'flex', flexDirection: 'column', position: 'relative', overflow: 'hidden' }}>
            <div style={{ padding: '20px 24px', position: 'relative', zIndex: 10 }}>
                <h2 style={{ margin: 0, fontSize: '24px', fontWeight: 800 }}>{t('board.title')}</h2>
                <div style={{ fontSize: '13px', color: '#666' }}>{t('board.subtitle')}</div>

                <div style={{ display: 'flex', gap: '8px', marginTop: '16px' }}>
                    <button
                        onClick={() => setMode(mode === 'vision' ? 'work' : 'vision')}
                        style={{ padding: '8px 16px', borderRadius: '12px', border: 'none', background: '#F0F2F5', fontWeight: 600, fontSize: '12px', cursor: 'pointer' }}
                    >
                        {mode === 'vision' ? `🎨 ${t('board.mode_vision')}` : `🏗️ ${t('board.mode_work')}`}
                    </button>
                    <button
                        onClick={() => setIsEditMode(!isEditMode)}
                        style={{ padding: '8px 16px', borderRadius: '12px', border: 'none', background: isEditMode ? theme.colors.ui.primary : '#eee', color: isEditMode ? '#fff' : '#000', fontWeight: 600, fontSize: '12px', cursor: 'pointer' }}
                    >
                        {isEditMode ? '✨ Done' : '⚙️ Edit'}
                    </button>
                    <button
                        onClick={() => setElements([])}
                        style={{ padding: '8px 16px', borderRadius: '12px', border: 'none', background: '#FFF5F5', color: '#FF4D4D', fontWeight: 600, fontSize: '12px', cursor: 'pointer', marginLeft: 'auto' }}
                    >
                        {t('board.clear')}
                    </button>
                </div>
            </div>

            <div
                ref={boardRef}
                style={{
                    flex: 1,
                    position: 'relative',
                    background: mode === 'vision' ? '#F8FAFC' : '#F1F5F9',
                    backgroundImage: mode === 'work' ? 'radial-gradient(#CBD5E1 1px, transparent 1px)' : 'none',
                    backgroundSize: '24px 24px',
                    overflow: 'auto',
                    touchAction: 'none'
                }}
            >
                {elements.length === 0 && (
                    <div style={{ position: 'absolute', inset: 0, display: 'flex', alignItems: 'center', justifyContent: 'center', color: '#94A3B8', textAlign: 'center', padding: '40px' }}>
                        <div>
                            <div style={{ fontSize: '48px', marginBottom: '12px' }}>☁️</div>
                            <div style={{ fontSize: '14px', fontWeight: 500 }}>{t('board.empty')}</div>
                        </div>
                    </div>
                )}

                {elements.map(el => (
                    <BoardElement
                        key={el.id}
                        element={el}
                        isEditMode={isEditMode}
                        onUpdate={updateElement}
                        onDelete={deleteElement}
                    />
                ))}
            </div>

            <AnimatePresence>
                {isEditMode && (
                    <motion.div
                        initial={{ y: 100 }}
                        animate={{ y: 0 }}
                        exit={{ y: 100 }}
                        style={{
                            position: 'absolute',
                            bottom: 20,
                            left: '50%',
                            transform: 'translateX(-50%)',
                            background: 'rgba(255,255,255,0.9)',
                            backdropFilter: 'blur(10px)',
                            padding: '12px 20px',
                            borderRadius: '24px',
                            boxShadow: '0 10px 25px rgba(0,0,0,0.1)',
                            display: 'flex',
                            gap: '12px',
                            zIndex: 100,
                            border: '1px solid #eee'
                        }}
                    >
                        <input type="file" id="board-img-upload" accept="image/*" style={{ display: 'none' }} onChange={handleImageUpload} />
                        <label htmlFor="board-img-upload" style={{ display: 'flex', alignItems: 'center', gap: '6px', cursor: 'pointer', background: theme.colors.ui.primary, color: '#fff', padding: '10px 16px', borderRadius: '16px', fontSize: '13px', fontWeight: 600 }}>
                            🖼️ {t('board.add_image')}
                        </label>
                        <button
                            onClick={() => {
                                const txt = prompt(t('board.add_note'));
                                if (txt) addElement(ELEMENT_TYPES.NOTE, txt);
                            }}
                            style={{ display: 'flex', alignItems: 'center', gap: '6px', cursor: 'pointer', background: '#fff', border: '1px solid #eee', padding: '10px 16px', borderRadius: '16px', fontSize: '13px', fontWeight: 600 }}
                        >
                            📝 {t('board.add_note')}
                        </button>
                    </motion.div>
                )}
            </AnimatePresence>

            {/* Minimal Dashboard Preview Logic Placeholder */}

import React, { useState } from 'react';
import { useTranslation } from 'react-i18next';
import { theme } from '../theme';

const RegularCalculator = () => {
    const { t } = useTranslation();
    const [input, setInput] = useState('');
    const [result, setResult] = useState('');

    const handleCalc = () => {
        try {
            // Safer evaluation using Function constructor
            // eslint-disable-next-line no-new-func
            const res = new Function('return ' + input)();
            setResult(String(res));
        } catch {
            setResult('Error');
        }
    };

    return (
        <div style={{ textAlign: 'center' }}>
            <div style={{ background: '#f5f5f5', padding: '20px', borderRadius: '16px', marginBottom: '16px', textAlign: 'right', fontSize: '24px', fontFamily: 'monospace', minHeight: '80px', display: 'flex', flexDirection: 'column', justifyContent: 'center' }}>
                <div style={{ color: '#999', fontSize: '14px', minHeight: '20px' }}>{result}</div>
                <div>{input || '0'}</div>
            </div>
            <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 1fr)', gap: '8px' }}>
                {['7', '8', '9', '/', '4', '5', '6', '*', '1', '2', '3', '-', 'C', '0', '=', '+'].map((btn) => (
                    <button
                        key={btn}
                        onClick={() => {
                            if (btn === '=') handleCalc();
                            else if (btn === 'C') { setInput(''); setResult(''); }
                            else setInput(prev => prev + btn);
                        }}
                        style={{ padding: '16px', borderRadius: '12px', border: 'none', background: '#fff', boxShadow: '0 2px 5px rgba(0,0,0,0.05)', fontSize: '18px', cursor: 'pointer', color: theme.colors.text.primary, transition: 'background 0.1s' }}
                        onMouseDown={e => e.currentTarget.style.background = '#eee'}
                        onMouseUp={e => e.currentTarget.style.background = '#fff'}
                    >
                        {btn}
                    </button>
                ))}
            </div>
        </div>
    );
};

const SavingsCalculator = () => {
    const { t } = useTranslation();
    const [initial, setInitial] = useState('1000');
    const [monthly, setMonthly] = useState('100');
    const [years, setYears] = useState('10');
    const [rate, setRate] = useState('5');

    const p = parseFloat(initial) || 0;
    const pm = parseFloat(monthly) || 0;
    const r = parseFloat(rate) / 100 / 12;
    const n = (parseFloat(years) || 0) * 12;

    const futureValue = p * Math.pow(1 + r, n) + (pm * (Math.pow(1 + r, n) - 1)) / r;
    const totalInvested = p + (pm * n);
    const totalInterest = futureValue - totalInvested;

    return (
        <div>
            <h3 style={{ marginTop: 0, fontSize: '18px' }}>{t('calc.mode_savings')}</h3>
            <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '12px', marginBottom: '20px' }}>
                <div>
                    <label style={{ display: 'block', fontSize: '11px', color: '#666', marginBottom: '4px' }}>{t('calc.initial')}</label>
                    <input value={initial} onChange={e => setInitial(e.target.value)} style={{ width: '100%', padding: '10px', borderRadius: '10px', border: '1px solid #ddd' }} />
                </div>
                <div>
                    <label style={{ display: 'block', fontSize: '11px', color: '#666', marginBottom: '4px' }}>{t('calc.monthly')}</label>
                    <input value={monthly} onChange={e => setMonthly(e.target.value)} style={{ width: '100%', padding: '10px', borderRadius: '10px', border: '1px solid #ddd' }} />
                </div>
                <div>
                    <label style={{ display: 'block', fontSize: '11px', color: '#666', marginBottom: '4px' }}>{t('calc.years')}</label>
                    <input value={years} onChange={e => setYears(e.target.value)} style={{ width: '100%', padding: '10px', borderRadius: '10px', border: '1px solid #ddd' }} />
                </div>
                <div>
                    <label style={{ display: 'block', fontSize: '11px', color: '#666', marginBottom: '4px' }}>{t('calc.rate')} %</label>
                    <input value={rate} onChange={e => setRate(e.target.value)} style={{ width: '100%', padding: '10px', borderRadius: '10px', border: '1px solid #ddd' }} />
                </div>
            </div>

            <div style={{ background: '#E8F5E9', padding: '20px', borderRadius: '20px', display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '16px' }}>
                <div style={{ textAlign: 'center' }}>
                    <div style={{ fontSize: '11px', color: '#2E7D32' }}>{t('calc.monthly_payment') || 'Ежемесячный платёж'}</div>
                    <div style={{ fontSize: '18px', fontWeight: 700, color: '#2E7D32' }}>${pm}</div>
                </div>
                <div style={{ textAlign: 'center' }}>
                    <div style={{ fontSize: '11px', color: '#2E7D32' }}>{t('calc.future_value') || 'Общая сумма'}</div>
                    <div style={{ fontSize: '18px', fontWeight: 700, color: '#2E7D32' }}>${Math.round(futureValue).toLocaleString()}</div>
                </div>
                <div style={{ textAlign: 'center' }}>
                    <div style={{ fontSize: '11px', color: '#2E7D32' }}>{t('calc.overpayment') || 'Переплата'}</div>
                    <div style={{ fontSize: '18px', fontWeight: 700, color: '#2E7D32' }}>${Math.round(totalInterest).toLocaleString()}</div>
                </div>
                <div style={{ textAlign: 'center' }}>
                    <div style={{ fontSize: '11px', color: '#2E7D32' }}>{t('calc.eff_rate') || 'Эффект. ставка'}</div>
                    <div style={{ fontSize: '18px', fontWeight: 700, color: '#2E7D32' }}>{rate}%</div>
                </div>
            </div>
        </div>
    );
};

export default function HistoryModal({ isOpen, onClose, title, goal }) {

    const [period, setPeriod] = useState('week'); // week, month, year, custom
    const [startDate, setStartDate] = useState(format(subDays(new Date(), 30), 'yyyy-MM-dd'));
    const [endDate, setEndDate] = useState(format(new Date(), 'yyyy-MM-dd'));

    const chartData = useMemo(() => {
        if (!goal) return null;

        const start = period === 'custom' ? new Date(startDate) : null;
        const end = period === 'custom' ? new Date(endDate) : new Date();

        let labels = [];
        const now = new Date();

        if (period === 'week') {
            for (let i = 6; i >= 0; i--) {
                labels.push(format(subDays(now, i), 'EEE'));
            }
        } else if (period === 'month') {
            for (let i = 29; i >= 0; i -= 2) { // More detailed for month
                labels.push(format(subDays(now, i), 'd MMM'));
            }
        } else if (period === 'year') {
            labels = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
        } else if (period === 'custom') {
            // Simplified custom labels
            labels = [format(start, 'd MMM'), '...', format(end, 'd MMM')];
        }

        const dataMap = {};
        goal.steps.forEach(s => {
            if (s.completed && s.completedAt) {
                const date = new Date(s.completedAt);

                // Filter by period
                let inRange = false;
                if (period === 'week') inRange = isAfter(date, subDays(now, 7));
                else if (period === 'month') inRange = isAfter(date, subDays(now, 30));
                else if (period === 'year') inRange = isAfter(date, startOfYear(now));
                else inRange = date >= start && date <= end;

                if (inRange) {
                    let dateKey;
                    if (period === 'year') dateKey = format(date, 'MMM');
                    else if (period === 'month') dateKey = format(date, 'd MMM');
                    else if (period === 'week') dateKey = format(date, 'EEE');
                    else dateKey = format(date, 'd MMM');

                    dataMap[dateKey] = (dataMap[dateKey] || 0) + 1;
                }
            }
        });

        const displayData = labels.map(l => dataMap[l] || 0);

        return {
            labels,
            datasets: [
                {
                    label: t('stats.xp'),
                    data: displayData,
                    fill: true,
                    backgroundColor: 'rgba(99, 102, 241, 0.1)',
                    borderColor: theme.colors.ui.primary,
                    borderWidth: 3,
                    tension: 0.4,
                    pointRadius: period === 'year' ? 2 : 4,
                    pointBackgroundColor: '#fff',
                    pointBorderColor: theme.colors.ui.primary,
                    pointBorderWidth: 2,
                },
            ],
        };
    }, [goal, period, startDate, endDate, t]);

    const options = {
        responsive: true,

        plugins: {

            y: {
                beginAtZero: true,
                grid: { display: true, color: '#f5f5f5', drawBorder: false },
                ticks: { stepSize: 1, color: '#999' }
            },
            x: {
                grid: { display: false },
                ticks: { color: '#999' }
            }
        },
    };

    const historyList = useMemo(() => {
        const list = [];
        if (goal) {
            goal.steps.forEach(s => {

                    list.push({
                        goalTitle: goal.title,
                        stepTitle: s.title,
                        date: s.completedAt,

    }, [goal]);

                    {['week', 'month', 'year', 'custom'].map(p => (
                        <button
                            key={p}

                            style={{
                                padding: '8px 16px',
                                borderRadius: '12px',
                                border: 'none',

export const getDailyQuote = () => {
    const now = new Date();
    const start = new Date(now.getFullYear(), 0, 0);
    const diff = now - start;
    const oneDay = 1000 * 60 * 60 * 24;
    const dayOfYear = Math.floor(diff / oneDay);
    return quotes[dayOfYear % quotes.length];
};

const CircularProgress = ({ percentage, size, strokeWidth, color, label }) => {
  const radius = (size - strokeWidth) / 2;
  const circumference = radius * 2 * Math.PI;
  const offset = circumference - (percentage / 100) * circumference;

  return (
    <div style={{ position: 'relative', width: size, height: size, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
      <svg width={size} height={size} style={{ transform: 'rotate(-90deg)' }}>
        <circle stroke="#f0f0f0" strokeWidth={strokeWidth} fill="transparent" r={radius} cx={size / 2} cy={size / 2} />
        <circle
          stroke={color}
          strokeWidth={strokeWidth}
          fill="transparent"
          strokeDasharray={circumference}
          strokeDashoffset={offset}
          strokeLinecap="round"
          r={radius}
          cx={size / 2}
          cy={size / 2}
          style={{ transition: 'stroke-dashoffset 0.5s ease' }}
        />
      </svg>
      <div style={{ position: 'absolute', textAlign: 'center' }}>
        {label && <div style={{ fontSize: '12px', fontWeight: 600, color: '#2d2d2d' }}>{label}</div>}
        <div style={{ fontSize: label ? '14px' : '16px', fontWeight: 700, color: '#2d2d2d' }}>{percentage}%</div>
      </div>
    </div>
  );
};

const BlobWidget = ({ width, height, shape, children, onClick, style, className, onLongPress, onPinchResize }) => {
  const [pressTimer, setPressTimer] = useState(null);
  const [initialDist, setInitialDist] = useState(null);

  const startPress = (e) => {
    if (e.touches && e.touches.length === 2) {
      const d = Math.hypot(e.touches[0].pageX - e.touches[1].pageX, e.touches[0].pageY - e.touches[1].pageY);
      setInitialDist(d);
      return;
    }
    const timer = setTimeout(() => {
      if (onLongPress) onLongPress();
    }, 600);
    setPressTimer(timer);
  };

  const handleTouchMove = (e) => {
    if (e.touches && e.touches.length === 2 && initialDist && onPinchResize) {
      const currentDist = Math.hypot(e.touches[0].pageX - e.touches[1].pageX, e.touches[0].pageY - e.touches[1].pageY);
      const ratio = currentDist / initialDist;
      if (ratio > 1.3) onPinchResize('large');
      else if (ratio < 0.7) onPinchResize('small');
    }
  };

  const endPress = () => {
    if (pressTimer) clearTimeout(pressTimer);
    setInitialDist(null);
  };

  return (
    <div
      onClick={onClick}
      onTouchStart={startPress}
      onTouchMove={handleTouchMove}
      onTouchEnd={endPress}
      onMouseDown={startPress}
      onMouseUp={endPress}
      onMouseLeave={endPress}
      className={className}
      style={{
        width: width || '100%',
        height: height || '100%',
        borderRadius: shape,
        background: '#ffffff',
        boxShadow: '0 10px 25px -5px rgba(0,0,0,0.05)',
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        justifyContent: 'center',
        cursor: 'pointer',
        transition: 'all 0.3s cubic-bezier(0.4, 0, 0.2, 1)',
        overflow: 'hidden',
        position: 'relative',
        zIndex: 5,
        padding: '12px',
        ...style
      }}
    >
      {children}
    </div>
  );
};

                  </div>
                )}
                <div style={{ position: 'absolute', bottom: '6px', right: '10px', fontSize: '8px', color: '#ddd', display: 'flex', alignItems: 'center', gap: '3px' }}>
                  <div style={{ width: '3px', height: '3px', borderRadius: '50%', background: '#4CAF50' }}></div>
                  {randomPartner}
                </div>
              </BlobWidget>
            </div>

      </div>
    </div>

  const renderDetail = () => {
